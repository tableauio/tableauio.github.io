    Scalar list #

    A worksheet ItemConf in HelloWorld.xml:

    <?xml version="1.0" encoding="UTF-8"?>
<!--
<@TABLEAU>
    <Item Sheet="ItemConf" />
</@TABLEAU>

<ItemConf>
    <Item>[int32]</Item>
</ItemConf>
-->

<ItemConf>
    <Item>1</Item>
    <Item>2</Item>
    <Item>3</Item>
</ItemConf>

    Generated:

    hello_world.proto
    // --snip--
option (tableau.workbook) = {name:"HelloWorld.xml"};

message ItemConf {
  option (tableau.worksheet) = {name:"ItemConf"};

  repeated int32 item_list = 1 [(tableau.field) = {name:"Item"}];
}
    ItemConf.json
    {
    "itemList": [
        1,
        2,
        3
    ]
}
